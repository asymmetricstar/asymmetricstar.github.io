/* Klasör oluşturma ve Npm paketlerinin kurulumu */
$mkdir .rest-demo
$cd ./rest-demo
$npm install express --save
$npm install body-parser --save




/* main.js */
const express = require("express");
const app = express();
const bodyParser = require("body-parser");
const router = express.Router();
    app.use(bodyParser.json());
    app.use(bodyParser.urlencoded({ extended: true }));
        router.post("/qr", (request, response) => {
            console.log(request.body);
            response.send({ "Status": "Success", "Type": "POST", "Date": Date.now() });
        });               
    app.use("/", router);
    app.listen(3000, () => console.log("Demo Rest-api server is started.."));
